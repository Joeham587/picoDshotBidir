
#pragma once

$PROG_GEN_C

#if !PICO_NO_HARDWARE
static const struct pio_program ${PROG_NAME}_program = {
    .instructions = ${PROG_NAME},
    .length = ${PROG_NAME}_wrap - ${PROG_NAME}_wrap_target + 1,
    .origin = -1,
};

static inline pio_sm_config ${PROG_NAME}_program_get_default_config(uint offset) {
    pio_sm_config c = pio_get_default_sm_config();
    sm_config_set_wrap(&c, offset + ${PROG_NAME}_wrap_target, offset + ${PROG_NAME}_wrap);
    return c;
}

$PROG_INLINE_C

#endif
